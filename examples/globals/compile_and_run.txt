To compile and run the example use:
- nvc++ -acc -Minfo=all -c main.cpp -I/home/tarchi/MiMMO/include
- nvc++ -acc -Minfo=all -c allvars.cpp -I/home/tarchi/MiMMO/include
- nvc++ -acc -Minfo=all main.o allvars.o -L/home/tarchi/MiMMO/build -lmimmo -o main.x
- export LD_LIBRARY_PATH=/home/tarchi/MiMMO/build:$LD_LIBRARY_PATH
- ./main.x
